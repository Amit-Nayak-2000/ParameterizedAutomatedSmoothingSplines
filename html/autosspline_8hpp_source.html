95 void smooth(){
96 //Work arrays
97 std::vector<std::vector<double>> V(npoint, std::vector<double>(7));
98 std::vector<double> A3(npoint);
99 std::vector<double> A2(npoint);
100 std::vector<double> A0(npoint);
101
102 this->setupq(dy, V, A3);
103 this->chol1d(V, A3, A2, A0);
104
105 for(int i = 0; i < npoint; i++){
106 A[i][0] = y[i] - 6 * (1 - p) * pow(dy[i], 2) * A0[i];
107 }
108
109 for(int i = 0; i < npoint; i++){
110 A[i][2] = A2[i] * 6 * p;
111 }
112
113 for(int i = 0; i < npoint - 1; i++){
114 A[i][3] = (A[i+1][2] - A[i][2]) / V[i][3];
115 A[i][1] = (A[i+1][0] - A[i][0]) / V[i][3] - (A[i][2] + A[i][3] / 3*V[i][3]) / 2*V[i][3];

125 void setupq(std::vector<double> &dx, std::vector<std::vector<double>> &V, std::vector<double> &QTY){
126
127
128 for(int i = 0; i < npoint - 1; i++){
129 V[i][3] = x[i+1] - x[i];
130 }
131
132 for(int i = 1; i < npoint - 1; i++){
133 V[i][0] = dx[i-1] / V[i-1][3];
134 }
135
136 V[npoint - 1][0] = 0.0;
137
138 for(int i = 1; i < npoint - 1; i++){
139 V[i][1] = -dx[i] / V[i][3] - dx[i] / V[i-1][3];
140 }
141
142 for(int i = 1; i < npoint - 1; i++){
143 V[i][2] = dx[i+1] / V[i][3];
144 }
145
146 for(int i = 1; i < npoint - 1; i++){
147 V[i][4] = pow(V[i][0], 2) + pow(V[i][1], 2) + pow(V[i][2], 2);
148 }
149
150 for(int i = 1; i < npoint - 2; i++){
151 V[i][5] = V[i][1] * V[i+1][0] + V[i][2] * V[i+1][1];
152 }
153
154 V[npoint - 2][5] = 0.0;
155
156 for(int i = 1; i < npoint - 3; i++){
157 V[i][6] = V[i][2] * V[i+2][0];
158 }
159
160 V[npoint - 3][6] = 0.0;
161 V[npoint - 2][6] = 0.0;
162
163 double prev = (y[1] - y[0]) / V[0][3];
164 double diff;
165
166 for(int i = 1; i < npoint - 1; i++){
167 diff = (y[i+1] - y[i]) / V[i][3];
168 QTY[i] = diff - prev;
169 prev = diff;
170 }

180 void chol1d(std::vector<std::vector<double>> &V, std::vector<double> &QTY, std::vector<double> &U, std::vector<double> &QU){
181
182 double six1mp = 6*(1-p);
183 double twop = 2*p;
184
185 for(int i = 1; i < npoint - 1; i++){
186 V[i][0] = six1mp * V[i][4] + twop*(V[i-1][3] + V[i][3]);
187 }
188
189 for(int i = 1; i < npoint - 1; i++){
190 V[i][1] = six1mp * V[i][5] + p*V[i][3];
191 }
192
193 for(int i = 1; i < npoint - 1; i++){
194 V[i][2] = six1mp * V[i][6];
195 }
196
197 if(npoint < 4){
198 U[0] = 0.0;
199 U[1] = QTY[1] / V[1][0];
200 U[2] = 0.0;
201 }
202 else{
203
204 double ratio;
205
206 for(int i = 1; i < npoint - 2; i++){
207 ratio = V[i][1] / V[i][0];
208 V[i+1][0] = V[i+1][0] - ratio * V[i][1];
209 V[i+1][1] = V[i+1][1] - ratio * V[i][2];
210 V[i][1] = ratio;
211 ratio = V[i][2] / V[i][0];
212 V[i+2][0] = V[i+2][0] - ratio*V[i][2];
213 V[i][2] = ratio;
214 }
215
216 U[0] = 0.0;
217 V[0][2] = 0.0;
218 U[1] = QTY[1];
219
220
221 for(int i = 1; i < npoint - 2; i++){
222 U[i+1] = QTY[i+1] - V[i][1] * U[i] - V[i-1][2] * U[i-1];
223 }
224
225 U[npoint - 1] = 0.0;
226 U[npoint - 2] = U[npoint - 2] / V[npoint - 2][0];
227
228 for(int i = npoint - 3; i >= 1; i--){
229 U[i] = U[i] / V[i][0] - U[i+1]*V[i][1] - U[i+2]*V[i][2];
230 }
231 }
232
233 double prev = 0.0;
234
235 for(int i = 1; i < npoint; i++){
236 QU[i] = (U[i] - U[i-1]) / V[i-1][3];
237 QU[i-1] = QU[i] - prev;
238 prev = QU[i];
239 }
240
241 QU[npoint - 1] = -QU[npoint - 1];
242
243

253 int findbreakindex(const std::vector<double> &breaks, const double &x, int lower, int upper){
254 assert(lower < upper);
255
256 //if the last (largest) element of breaks is smaller than or equal to x.
257 if(breaks[upper] <= x){
258 return upper;
259 }
260
261 int middle = lower + (upper - lower) * 0.5;
262
263 if(breaks[middle] <= x && breaks[middle + 1] > x){
264 return middle;
265 }
266 //if the middle element is smaller than x, the appropriate index lies between middle + 1 and upper
267 else if(breaks[middle + 1] < x){
268 return findbreakindex(breaks, x, middle + 1, upper);
269 }
270 //middle element is greater than x, appropriate index lies between lower and middle -1
271 else{
272 return findbreakindex(breaks, x, lower, middle);
273 }
274
275 //if could not find index
276 return -1;
277
278 }

288 double ppvalu(const std::vector<double> &breaks, std::vector<std::vector<double>> &coeff, double eval, int jderiv, int K){
289 int L = breaks.size() - 1;
290 double result = 0.0;
291
292 double fmmjdr = K - jderiv;
293
294 if(fmmjdr <= 0){
295 return result;
296 }
297
298 //returns c style index
299 int i = findbreakindex(breaks, eval, 0, L);
300
301 assert(i != -1);
302
303 double h = eval - breaks[i];
304
305 int m = K - 1;
306
307
308 bool flag = true;
309
310
311 while(flag){
312 result = ((result / fmmjdr) * h) + coeff[m][i];
313 m--;
314 fmmjdr -= 1;
315
316 if(fmmjdr <= 0){
317 flag = false;
318 break;
319 }
320 }
321
322
323 return result;
324

377std::vector<double> parametrize(const std::vector<double> &x, const std::vector<double> &y){
378 int npoint = x.size();
379 double totalarclength = 0;
380
381 for(int i = 1; i < npoint; i++){
382 totalarclength += pow((pow((x[i] - x[i-1]), 2) + pow((y[i] - y[i-1]), 2)), 0.5);
383 }
384
385 std::vector<double> result(npoint);
386 result[0] = 0;
387
388 for(int i = 1; i < npoint; i++){
389 result[i] = (result[i-1] + pow((pow((x[i] - x[i-1]), 2) + pow((y[i] - y[i-1]), 2)), 0.5));
390 }
391
392 for(int i = 0; i < npoint; i++){
393 result[i] = result[i] / totalarclength;
394 }
395
396 return result;
397}

424void numfirstder(const std::vector<double> &x, const std::vector<double> &y, std::vector<double> &yprime){
425 int N = x.size();
426
427 for(int i = 0; i < N; i++){
428 //beginning edge case
429 //forward difference approximation
430 if(i == 0){
431 yprime[i] = (y[i+1] - y[i]) / (x[i+1] - x[i]);
432 }
433 //ending edge case
434 //backward difference approximation
435 else if (i == (N - 1)){
436 yprime[i] = (y[i] - y[i-1]) / (x[i] - x[i-1]);
437 }
438 //centered difference approximation
439 else{
440 yprime[i] = (y[i+1] - y[i-1]) / (x[i+1] - x[i-1]);
441 }
442 }
443

452void num2der(const std::vector<double> &x, const std::vector<double> &y, std::vector<double> &ydoubleprime){
453 int N = x.size();
454
455 for(int i = 0; i < N; i++){
456 //beginning edge case
457 //forward difference approximation
458 if(i == 0){
459 ydoubleprime[i] = (2*y[i] - 5*y[i+1] + 4*y[i+2] - y[i+3]) / pow((x[i+1] - x[i]), 2);
460 }
461 //ending edge case
462 //backward difference approximation
463 else if (i == (N - 1)){
464 ydoubleprime[i] = (2*y[i] - 5*y[i-1] + 4*y[i-2] - y[i-3]) / pow((x[i] - x[i-1]), 2);
465 }
466 //centered difference approximation
467 else{
468 ydoubleprime[i] = (y[i+1] - 2*y[i] + y[i-1]) / pow((x[i+1] - x[i]), 2);
469 }
470 }

481void radiusofcurve(const std::vector<double> &t, const std::vector<double> &x, const std::vector<double> &y, std::vector<double> &result){
482 int N = t.size();
483
484 std::vector<double> xprime(N);
485 std::vector<double> yprime(N);
486 std::vector<double> xdoubleprime(N);
487 std::vector<double> ydoubleprime(N);
488
489 numfirstder(t, x, xprime);
490 numfirstder(t, y, yprime);
491 num2der(t, x, xdoubleprime);
492 num2der(t, y, ydoubleprime);
493
494
495 double denominator, zerolim;
496 zerolim = 10*std::numeric_limits<double>::epsilon();
497
498 for(int i = 0; i < N; i++){
499 denominator = std::abs(xprime[i]*ydoubleprime[i] - yprime[i]*xdoubleprime[i]);
500
501 //if denominator is close to 0, set radius of curvature to max value.
502 if(denominator <= zerolim){
503 result[i] = std::numeric_limits<double>::max();
504 continue;
505 }
506
507 result[i] = std::abs(pow((pow(xprime[i], 2) + pow(yprime[i], 2)),1.5)) / denominator;
508 }

719 void RunBisecant(){
720 //middle point for bisection
721 double p = (u + l) / 2;
722
723 eps *= 100;
724
725 int evalsize = tvals.size();
726
727 //construct inital splines for x and y
728 Xspline = SSpline(t, x, p);
729 Yspline = SSpline(t, y, p);
730
731 //evaluate initial spline points
732 Xspline.evalSpline(tvals, xnew);
733 Yspline.evalSpline(tvals, ynew);
734
735 //vector to contain radius of curvature
736 std::vector<double> radius(evalsize);
737
738 double M;
739
740 //Bisection begins here
741 for(int i = 0; i < 150; i++){
742 //get radius of curvature
743 radiusofcurve(tvals, xnew, ynew, radius);
744
745 //get minimum radius of curvature
746 M = *std::min_element(radius.begin(), radius.end());
747
748 //check if obtained min radius of curvature satisfies the user requirement.
749 if(std::abs(M - userdef) < eps){
750 std::cout << "Bisection finished with " << i+1 << " iterations." << std::endl;
751 std::cout << std::endl;
752 fp = p;
753 result = M;
754
755 //call secant to finish iteration to find optimal smoothing parameter and smooth data.
756 secant();
757
758 return;
759 }
760 else if (M < userdef){
761 //recalculate smoothing param
762 u = p;
763 p += 0.5*(l - p);
764
765 //recalculate splines with new smoothing param
766 Xspline.changeSmooth(p);
767 Yspline.changeSmooth(p);
768 Xspline.evalSpline(tvals, xnew);
769 Yspline.evalSpline(tvals, ynew);
770 }
771 else if (M > userdef){
772 //recalculate smoothing param
773 l = p;
774 p += 0.5*(u - p);
775
776 //recalculate splines with new smoothing param
777 Xspline.changeSmooth(p);
778 Yspline.changeSmooth(p);
779 Xspline.evalSpline(tvals, xnew);
780 Yspline.evalSpline(tvals, ynew);
781 }
782
783 fp = p;
784 result = M;
785
786 }

797 void secant(){
798 double a = l;
799 double b = u;
800 eps /= 100;
801
802 //construct inital splines for a
803 Xspline.changeSmooth(a);
804 Yspline.changeSmooth(a);
805
806
807 //evaluate initial spline points
808 Xspline.evalSpline(tvals, xnew);
809 Yspline.evalSpline(tvals, ynew);
810
811 double fa = secantFunc();
812
813 //change spline param to B
814 Xspline.changeSmooth(b);
815 Yspline.changeSmooth(b);
816 Xspline.evalSpline(tvals, xnew);
817 Yspline.evalSpline(tvals, ynew);
818
819 double fb = secantFunc();
820
821 double d;
822
823 //Secant method begins here
824 for(int i = 0; i <= 150; i++){
825 d = ((b - a) / (fb - fa));
826 b = a;
827 fb = fa;
828 d = d*fa;
829 a = a - d;
830
831 //evaluate new a
832 Xspline.changeSmooth(a);
833 Yspline.changeSmooth(a);
834 Xspline.evalSpline(tvals, xnew);
835 Yspline.evalSpline(tvals, ynew);
836
837 //call helper function which provides the difference of the obtained and user defined min radius of curvature.
838 fa = secantFunc();
839
840 if(std::abs(fa) <= eps){
841 std::cout << "Secant finished with " << i+1 << " iterations." << std::endl;
842 std::cout << std::endl;
843 fp = a;
844 result = fa + userdef;
845 std::cout << "Final radius of curvature is " << result << "." << std::endl;
846 return;
847 }
